%
% sequence generator [SEP] {LIST} [FUNC]
%
\newcommandx*{\mkseq}[3][1=\, , 3=\idfunc]{%
	\foreach \elem [count=\i] in {#2} {%
		\ifnum\i=1%
			#3{\elem}%
		\else%
			#1 #3{\elem}%
		\fi%
	}%
}%
